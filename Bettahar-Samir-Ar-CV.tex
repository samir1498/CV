% Redefine cventry to match user preference:
% Single column, ALL elements aligned to the Right (R = raggedleft = Physical Right)
\renewcommand*{\cventry}[5]{%

  \setlength\tabcolsep{0pt}
  \setlength{\extrarowheight}{0pt}
  \begin{tabular*}{\textwidth}{@{\extracolsep{\fill}} R{\textwidth}}
    \entrytitlestyle{#2} \\
    \entrypositionstyle{#1} \\
    {\entrylocationstyle{#3} \ \textbullet\ \entrydatestyle{#4}} \\
    \descriptionstyle{#5}
  \end{tabular*}%
}
